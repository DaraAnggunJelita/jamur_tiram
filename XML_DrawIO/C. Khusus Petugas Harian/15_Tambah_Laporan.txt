@startuml
hide footbox
skinparam monochrome true
skinparam shadowing false
actor "Petugas" as Petugas
boundary "Halaman Tambah Laporan" as B_Tambah
control "Proses" as C_Proses
entity "Tabel Laporan" as E_Laporan
boundary "Halaman Data Laporan" as B_Data

activate Petugas
Petugas -> B_Tambah : Akses halaman tambah laporan()
activate B_Tambah
Petugas -> B_Tambah : Input data laporan()
B_Tambah -> C_Proses : Kirim data()
activate C_Proses
C_Proses -> C_Proses : Validasi()
activate C_Proses
deactivate C_Proses
C_Proses -> E_Laporan : Simpan data()
activate E_Laporan
deactivate E_Laporan
C_Proses --> B_Tambah : Gagal()
deactivate B_Tambah
C_Proses -> B_Data : Berhasil()
activate B_Data
deactivate B_Data
deactivate C_Proses
deactivate Petugas
@enduml
