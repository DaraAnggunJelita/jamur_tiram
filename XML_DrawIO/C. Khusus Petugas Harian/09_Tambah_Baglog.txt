@startuml
hide footbox
skinparam monochrome true
skinparam shadowing false
actor "Petugas" as Petugas
boundary "Halaman Tambah Baglog" as B_Tambah
control "Proses" as C_Proses
entity "Tabel Baglogs" as E_Baglogs
boundary "Halaman Data Baglog" as B_Data

activate Petugas
Petugas -> B_Tambah : Akses halaman tambah baglog()
activate B_Tambah
Petugas -> B_Tambah : Input data baglog()
B_Tambah -> C_Proses : Kirim data()
activate C_Proses
C_Proses -> C_Proses : Validasi()
activate C_Proses
deactivate C_Proses
C_Proses -> E_Baglogs : Simpan data()
activate E_Baglogs
deactivate E_Baglogs
C_Proses --> B_Tambah : Gagal()
deactivate B_Tambah
C_Proses -> B_Data : Berhasil()
activate B_Data
deactivate B_Data
deactivate C_Proses
deactivate Petugas
@enduml
