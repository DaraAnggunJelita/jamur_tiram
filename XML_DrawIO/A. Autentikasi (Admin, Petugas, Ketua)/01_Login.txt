@startuml
hide footbox
skinparam monochrome true
skinparam shadowing false
actor "Admin" as Admin
boundary "Halaman Login" as B_Login
control "Proses" as C_Proses
entity "Tabel Users" as E_Users
boundary "Halaman Dashboard" as B_Dash

activate Admin
Admin -> B_Login : Akses halaman login()
activate B_Login
Admin -> B_Login : Input email, password()
B_Login -> C_Proses : Kirim data()
activate C_Proses
C_Proses -> E_Users : Cek data()
activate E_Users
deactivate E_Users
C_Proses -> C_Proses : Validasi()
activate C_Proses
deactivate C_Proses
C_Proses --> B_Login : Gagal()
deactivate B_Login
C_Proses -> B_Dash : Berhasil()
activate B_Dash
deactivate B_Dash
deactivate C_Proses
deactivate Admin
@enduml
