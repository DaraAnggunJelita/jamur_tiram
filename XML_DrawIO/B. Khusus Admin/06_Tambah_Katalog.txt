@startuml
hide footbox
skinparam monochrome true
skinparam shadowing false
actor "Admin" as Admin
boundary "Halaman Tambah Katalog" as B_Tambah
control "Proses" as C_Proses
entity "Tabel Catalogs" as E_Catalogs
boundary "Halaman Data Katalog" as B_Data

activate Admin
Admin -> B_Tambah : Akses halaman tambah katalog()
activate B_Tambah
Admin -> B_Tambah : Input data katalog()
B_Tambah -> C_Proses : Kirim data()
activate C_Proses
C_Proses -> C_Proses : Validasi()
activate C_Proses
deactivate C_Proses
C_Proses -> E_Catalogs : Simpan data()
activate E_Catalogs
deactivate E_Catalogs
C_Proses --> B_Tambah : Gagal()
deactivate B_Tambah
C_Proses -> B_Data : Berhasil()
activate B_Data
deactivate B_Data
deactivate C_Proses
deactivate Admin
@enduml
