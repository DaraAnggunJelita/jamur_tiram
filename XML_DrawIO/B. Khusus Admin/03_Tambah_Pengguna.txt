@startuml
hide footbox
skinparam monochrome true
skinparam shadowing false
actor "Admin" as Admin
boundary "Halaman Tambah Pengguna" as B_Tambah
control "Proses" as C_Proses
entity "Tabel Users" as E_Users
boundary "Halaman Data Pengguna" as B_Data

activate Admin
Admin -> B_Tambah : Akses halaman tambah pengguna()
activate B_Tambah
Admin -> B_Tambah : Input data pengguna()
B_Tambah -> C_Proses : Kirim data()
activate C_Proses
C_Proses -> C_Proses : Validasi()
activate C_Proses
deactivate C_Proses
C_Proses -> E_Users : Simpan data()
activate E_Users
deactivate E_Users
C_Proses --> B_Tambah : Gagal()
deactivate B_Tambah
C_Proses -> B_Data : Berhasil()
activate B_Data
deactivate B_Data
deactivate C_Proses
deactivate Admin
@enduml
