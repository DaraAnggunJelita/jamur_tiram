actor "Admin" as Admin
boundary "Halaman Edit Pengguna" as B_Edit
control "Proses" as C_Proses
entity "Tabel Users" as E_Users
boundary "Halaman Data Pengguna" as B_Data

activate Admin
Admin -> B_Edit : Akses halaman edit pengguna()
activate B_Edit
Admin -> B_Edit : Input data baru()
B_Edit -> C_Proses : Kirim data()
activate C_Proses
C_Proses -> C_Proses : Validasi()
activate C_Proses
deactivate C_Proses
C_Proses -> E_Users : Update data()
activate E_Users
deactivate E_Users
C_Proses --> B_Edit : Gagal()
deactivate B_Edit
C_Proses -> B_Data : Berhasil()
activate B_Data
deactivate B_Data
deactivate C_Proses
deactivate Admin
@enduml
