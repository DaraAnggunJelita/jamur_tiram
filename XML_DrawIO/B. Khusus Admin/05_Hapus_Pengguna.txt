@startuml
hide footbox
skinparam monochrome true
skinparam shadowing false
actor "Admin" as Admin
boundary "Halaman Data Pengguna" as B_Data
control "Proses" as C_Proses
entity "Tabel Users" as E_Users

activate Admin
Admin -> B_Data : Klik hapus()
activate B_Data
B_Data -> C_Proses : Kirim request()
activate C_Proses
C_Proses -> E_Users : Hapus data()
activate E_Users
deactivate E_Users
C_Proses --> B_Data : Gagal()
C_Proses -> B_Data : Berhasil()
deactivate B_Data
deactivate C_Proses
deactivate Admin
@enduml
